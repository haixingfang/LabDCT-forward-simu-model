plot

(method of ODF)

plots odf

Syntax 

% plot the odf along the alpha fibre
f = fibre.alpha(odf.CS)
plot(odf,f)
% plot the odf as phi2 sections
oS = phi2Sections(odf.CS)
plot(odf,oS)

this function is only a shortcut to plotSection, plot3d and plotFibre
